花括号{}(Curly Braces)

// 糟糕的实例
for (var i = 0; i < 10; i += 1)
   alert(i);
// 糟糕的实例
for (var i = 0; i < 10; i += 1)
   alert(i);
   alert(i + " is " + (i % 2 ? "odd" : "even"));
// 好的实例
for (var i = 0; i < 10; i += 1) {
   alert(i);
}
